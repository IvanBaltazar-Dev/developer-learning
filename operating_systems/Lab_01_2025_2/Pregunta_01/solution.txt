#!/bin/bash
set -euo pipefail

prefijo="file"

for path in "$prefijo"*
do
   fname="${path##*/}"
   sufijo="${fname#"$prefijo"}"

   if [[ ! "$sufijo" =~ ^[0-9]+* ]]
   then
      continue
   fi

   printf -v padded "%02d" "${sufijo}"

   new_name="${prefijo}_${padded}"

   if [[ "${fname}" == "${new_name}" ]]
   then
      continue
   fi

   mv -- "$path" "${new_name}"

done
